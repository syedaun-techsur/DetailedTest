```javascript
require('dotenv').config();
const { Sequelize } = require('sequelize');

const databaseUrl = process.env.DATABASE_URL;
if (!databaseUrl) {
  throw new Error('DATABASE_URL environment variable is required. Set it to a postgres:// or sqlite:/// connection string.');
}

// Parse dialect from URL scheme: postgres:// → postgres, sqlite:// → sqlite
const isSQLite = databaseUrl.startsWith('sqlite');

const sequelize = new Sequelize(databaseUrl, {
  dialect: isSQLite ? 'sqlite' : 'postgres',
  // SQLite: extract file path from sqlite:///./taskflow.db → ./taskflow.db
  storage: isSQLite ? databaseUrl.replace(/^sqlite:\/\/\//, '') : undefined,
  logging: process.env.NODE_ENV === 'development' ? console.log : false,
  pool: {
    max: 10,
    min: 2,
    acquire: 30000,
    idle: 10000,
  },
  dialectOptions: isSQLite ? {} : {
    // Recommended for PostgreSQL production (TechArch §5.6)
    // ssl: process.env.NODE_ENV === 'production' ? { require: true } : false,
  },
});

module.exports = sequelize;
```

Create the Task Sequelize model and the versioned migration. All column names, types, constraints, and indexes MUST match TechArch-TaskFlow.md §3.2 Complete DDL exactly.

**Canonical PostgreSQL DDL (from TechArch §3.2):**
```sql
CREATE TABLE tasks (
    id          SERIAL                      PRIMARY KEY,
    title       VARCHAR(255)                NOT NULL,
    description TEXT                        NULL,
    status      VARCHAR(20)                 NOT NULL
                    DEFAULT 'To Do'
                    CONSTRAINT tasks_status_check
                        CHECK (status IN ('To Do', 'In Progress', 'Done')),
    due_date    DATE                        NULL,
    created_at  TIMESTAMP WITH TIME ZONE    NOT NULL DEFAULT NOW(),
    updated_at  TIMESTAMP WITH TIME ZONE    NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_tasks_status     ON tasks (status);
CREATE INDEX idx_tasks_created_at ON tasks (created_at DESC);
```

**`src/models/Task.js`** — Sequelize model definition:
```javascript
const { DataTypes, Model } = require('sequelize');
const sequelize = require('../../config/database');

class Task extends Model {}

Task.init({
  id: {
    type: DataTypes.INTEGER,
    autoIncrement: true,
    primaryKey: true,
    allowNull: false,
  },
  title: {
    type: DataTypes.STRING(255),
    allowNull: false,
    validate: {
      notNull: { msg: 'Title is required.' },
      notEmpty: { msg: 'Title is required.' },
      len: { args: [1, 255], msg: 'Title must be 255 characters or fewer.' },
    },
  },
  description: {
    type: DataTypes.TEXT,
    allowNull: true,
    defaultValue: null,
  },
  status: {
    type: DataTypes.STRING(20),
    allowNull: false,
    defaultValue: 'To Do',
    validate: {
      isIn: {
        args: [['To Do', 'In Progress', 'Done']],
        msg: 'Status must be one of: To Do, In Progress, Done.',
      },
    },
  },
  due_date: {
    type: DataTypes.DATEONLY,   // DATE type — no time component
    allowNull: true,
    defaultValue: null,
  },
  // createdAt / updatedAt are managed automatically by Sequelize
  // (underscored: true maps camelCase to snake_case)
}, {
  sequelize,
  tableName: 'tasks',
  modelName: 'Task',
  underscored: true,          // maps createdAt → created_at, updatedAt → updated_at
  timestamps: true,           // automatically manage created_at and updated_at
});

module.exports = Task;
```

**`migrations/001-create-tasks.js`** — versioned migration (from TechArch §3.2 DDL):
```javascript
'use strict';

module.exports = {
  async up(queryInterface, Sequelize) {
    await queryInterface.createTable('tasks', {
      id: {
        type: Sequelize.INTEGER,
        autoIncrement: true,
        primaryKey: true,
        allowNull: false,
      },
      title: {
        type: Sequelize.STRING(255),
        allowNull: false,
      },
      description: {
        type: Sequelize.TEXT,
        allowNull: true,
        defaultValue: null,
      },
      status: {
        type: Sequelize.STRING(20),
        allowNull: false,
        defaultValue: 'To Do',
      },
      due_date: {
        type: Sequelize.DATEONLY,
        allowNull: true,
        defaultValue: null,
      },
      created_at: {
        type: Sequelize.DATE,
        allowNull: false,
        defaultValue: Sequelize.literal('CURRENT_TIMESTAMP'),
      },
      updated_at: {
        type: Sequelize.DATE,
        allowNull: false,
        defaultValue: Sequelize.literal('CURRENT_TIMESTAMP'),
      },
    });

    // Add CHECK constraint for status (PostgreSQL only — SQLite accepts but may not enforce)
    // Using raw SQL for the constraint since Sequelize's createTable doesn't support CHECK
    await queryInterface.sequelize.query(
      `ALTER TABLE tasks ADD CONSTRAINT tasks_status_check CHECK (status IN ('To Do', 'In Progress', 'Done'))`
    ).catch(() => {
      // SQLite does not support ADD CONSTRAINT via ALTER TABLE; constraint is included inline below
      // The ORM-level validation in the Task model enforces this on all writes
      console.log('Note: CHECK constraint via ALTER TABLE not supported on this DB dialect — ORM-level validation enforces status values.');
    });

    // Indexes (from TechArch §3.2)
    await queryInterface.addIndex('tasks', ['status'], {
      name: 'idx_tasks_status',
    });
    await queryInterface.addIndex('tasks', ['created_at'], {
      name: 'idx_tasks_created_at',
      order: 'DESC',
    });
  },

  async down(queryInterface, Sequelize) {
    await queryInterface.dropTable('tasks');
  },
};
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client→DB | ORM layer accepting application-controlled values and issuing SQL against the database |
| env→app | `DATABASE_URL` environment variable crossing into the Sequelize constructor (connection string with embedded credentials) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering (SQL injection) | `src/models/Task.js` — all ORM query methods (`Task.findAll`, `Task.create`, `Task.update`, `Task.destroy`) | mitigate | All queries use Sequelize ORM parameterized methods exclusively — never raw string-concatenated SQL. Sequelize uses prepared statements under the hood (TechArch §5.2). Raw SQL is used only in the migration for the CHECK constraint ALTER TABLE, which contains no user-supplied data. |
| T-02-02 | Information disclosure | `config/database.js` — `DATABASE_URL` credential in connection string | mitigate | `DATABASE_URL` is loaded from environment variable via `dotenv` in `config/database.js`; never hardcoded or committed. `.env` is listed in `.gitignore`. `.env.example` contains only a placeholder SQLite URL with no real credentials. |
| T-02-03 | Denial of service | `config/database.js` — connection pool | accept | Pool capped at `max: 10` connections (TechArch §7.2 specifies 5–10 for ≤15 concurrent users). Pool exhaustion under load beyond this is accepted; residual risk owned by the operations team, mitigated by network-layer access control (firewall/VPN). |
| T-02-04 | Elevation of privilege | `migrations/001-create-tasks.js` — `CURRENT_TIMESTAMP` default and `Sequelize.literal()` | mitigate | The only `Sequelize.literal()` call in the migration uses the developer-controlled string `'CURRENT_TIMESTAMP'` — not any user-supplied value. No user input reaches the migration. The migration runs only at deploy time under developer control, not at request time. |
</threat_model>
